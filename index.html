   //Insertion Sort Algorithm
   //Input: Array A[0..n-1]
   //Output: Array S[0..n-1] of A's elements sorted
    for j = 1 to A.length
            key = A[j]
            //Insert A[j] into the sorted sequence A[1...j-1]
            i = j - 1
            while i > 0 and A[i] > key
                        A[i+1] = A[i]
                        i = i - 1
            A[i + 1] = key
